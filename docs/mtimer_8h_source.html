13 // Struct providing access to RISC-V Steel MTimer Device registers
14 typedef struct
15 {
16  // Control Register (CR). Address offset: 0x00
17  volatile uint32_t CR;
18  // Machine Time (MTIME) Register lowest 32 bits. Address offset: 0x04
19  volatile uint32_t MTIMEL;
20  // Machine Time (MTIME) Register highest 32 bits. Address offset: 0x08
21  volatile uint32_t MTIMEH;
22  // Machine Time Compare (MTIMECMP) Register lowest 32 bits. Address offset: 0x0c
23  volatile uint32_t MTIMECMPL;
24  // Machine Time Compare (MTIMECMP) Register highest 32 bits. Address offset: 0x10
25  volatile uint32_t MTIMECMPH;
26 } MTimerController;

28 // Offset of the Counter Enable (EN) bit of the MTIME register
29 #define MTIMER_CR_EN_OFFSET 0U

31 // Mask used to read/set/clear the Counter Enable (EN) bit of the MTIME register
32 #define MTIMER_CR_EN_MASK (0x1U << MTIMER_CR_EN_OFFSET)

63 inline void mtimer_set_counter(MTimerController *mtimer, uint64_t new_value)
64 {
65  mtimer->MTIMEL = new_value & 0xFFFFFFFF;
66  mtimer->MTIMEH = new_value >> 32;
67 }

75 inline uint64_t mtimer_get_counter(MTimerController *mtimer)
76 {
77  uint32_t cnt_l = mtimer->MTIMEL;
78  uint64_t cnt_h = mtimer->MTIMEH;
79  return (cnt_h << 31) | cnt_l;
80 }

87 inline void mtimer_clear_counter(MTimerController *mtimer)
88 {
89  mtimer->MTIMEL = 0;
90  mtimer->MTIMEH = 0;
91 }

99 inline void mtimer_set_compare(MTimerController *mtimer, uint64_t new_value)
100 {
101  /* Writing -1 to MTIMECMPL, writing the MSB word first and finally writing the LSB word prevents
102  * spurious interrupts to be triggered due to the intermediate value held by the register during
103  * the update.
104  *
105  * See RISC-V Specifications, v.2 (privileged architecture) pp. 45-46 */
106  mtimer->MTIMECMPL = -1;
107  mtimer->MTIMECMPH = new_value >> 32;
108  mtimer->MTIMECMPL = new_value & 0xFFFFFFFF;
109 }
